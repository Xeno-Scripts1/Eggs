-- Helper function to create gradients
local function createGradient(frame, color1, color2)
    local gradient = Instance.new("UIGradient")
    gradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0, color1), ColorSequenceKeypoint.new(1, color2)}
    gradient.Parent = frame
    return gradient
end

-- Dragging logic
local dragging = false
local dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X,
                                   startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

-- Dupe pet function
local function dupePet()
    local equippedTool = Character:FindFirstChildOfClass("Tool")
    if not equippedTool then
        StartButton.Text = "⚠️ No pet equipped!"
        wait(1.5)
        StartButton.Text = "START DUPING"
        return
    end

    for _, child in pairs(Character:GetChildren()) do
        if child.Name:find("_VisualClone") then
            child:Destroy()
        end
    end

    local clone = equippedTool:Clone()
    clone.Name = equippedTool.Name .. "_VisualClone"

    local handle = clone:FindFirstChild("Handle") or clone:FindFirstChildWhichIsA("BasePart")
    if handle then
        handle.Anchored = false
        handle.CanCollide = false
        clone.Parent = Character

        local hand = Character:FindFirstChild("RightHand") or Character:FindFirstChild("Right Arm")
        if hand then
            local weld = Instance.new("WeldConstraint")
            weld.Part0 = hand
            weld.Part1 = handle
            weld.Parent = handle

            handle.CFrame = hand.CFrame * CFrame.new(0, -0.5, -1)
        end
    end

    StartButton.Text = "✅ Pet Duplicated!"
    wait(1.5)
    StartButton.Text = "START DUPING"
end

-- Animate loading bar
local function animateLoadingBar(duration)
    LoadingBarFill.Size = UDim2.new(0, 0, 1, 0)
    LoadingBarBackground.Visible = true
    LoadingPercent.Text = "0%"
    StartButton.AutoButtonColor = false
    StartButton.Active = false
    StartButton.Text = "⏳ Duplicating..."

    local startTime = tick()
    local connection

    connection = RunService.RenderStepped:Connect(function()
        local elapsed = tick() - startTime
        local progress = math.clamp(elapsed / duration, 0, 1)
        LoadingBarFill.Size = UDim2.new(progress, 0, 1, 0)
        LoadingPercent.Text = string.format("%d%%", math.floor(progress * 100))
        if progress >= 1 then
            connection:Disconnect()
        end
    end)

    wait(duration)

    LoadingBarBackground.Visible = false
    StartButton.AutoButtonColor = true
    StartButton.Active = true
    StartButton.Text = "START DUPING"
end

-- Start Button click
StartButton.MouseButton1Click:Connect(function()
    if not StartButton.Active then return end
    coroutine.wrap(function()
        animateLoadingBar(7)
        dupePet()
    end)()
end)
