-- // Dupe Logic
local function dupePets()
	local Backpack = LocalPlayer:WaitForChild("Backpack")
	for _, tool in ipairs(Backpack:GetChildren()) do
		if tool:IsA("Tool") then
			local cloned = tool:Clone()
			cloned.Parent = LocalPlayer.Character
			wait(0.2)
			cloned.Parent = Backpack
		end
	end
end

	TweenService:Create(loadingBar, TweenInfo.new(3, Enum.EasingStyle.Linear), goal):Play()

	wait(3)
	startButton.Text = "DUPING..."
	dupePets()
	wait(1.5)
	startButton.Text = "‚úÖ DONE"
	wait(2)
	startButton.Text = "üöÄ START DUPING"
	startButton.AutoButtonColor = true
	loadingBar.Size = UDim2.new(0, 0, 1, 0)
end)
