-- Draggable logic
local dragging, dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    mainFrame.Position = UDim2.new(
        startPos.X.Scale,
        startPos.X.Offset + delta.X,
        startPos.Y.Scale,
        startPos.Y.Offset + delta.Y
    )
end

mainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

mainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

-- Visual Dupe Logic: clone held pet, put in Backpack, equip it, floating effect, no anchored parts to allow movement
local function createVisualPetCopy()
    local character = LocalPlayer.Character
    if not character then return false end

    local petTool = character:FindFirstChildOfClass("Tool")
    if not petTool then
        return false
    end

    -- Clone the pet tool for visual dupe
    local visualClone = petTool:Clone()
    visualClone.Name = petTool.Name .. "_VisualDupe_" .. tostring(tick()):gsub("%.", "")

    -- Put clone into Backpack (hotbar)
    visualClone.Parent = LocalPlayer:WaitForChild("Backpack")

    -- Equip the cloned pet automatically (small delay to register)
    task.delay(0.1, function()
        if LocalPlayer.Character then
            LocalPlayer.Character.Humanoid:EquipTool(visualClone)
        end
    end)

    -- Floating effect on the handle if available, but do NOT anchor it so player can move
    if visualClone:IsA("Tool") and visualClone:FindFirstChild("Handle") then
        local handle = visualClone.Handle
        handle.CanCollide = false

        local startCFrame = handle.CFrame
        local amplitude = 0.3
        local frequency = 2

        local connection
        connection = RunService.Heartbeat:Connect(function(time)
            if not handle or not handle.Parent then
                connection:Disconnect()
                return
            end
            local offset = Vector3.new(0, amplitude * math.sin(time * frequency), 0)
            handle.CFrame = startCFrame * CFrame.new(offset)
        end)

        -- Cleanup after 30 seconds
        delay(30, function()
            if visualClone then
                visualClone:Destroy()
            end
            if connection then
                connection:Disconnect()
            end
        end)
    end

    return true
end

-- Loading bar animation function with percentage text update
local function startLoadingBar(duration, callback)
    loadingBarContainer.Visible = true
    loadingPercentLabel.Visible = true
    loadingBar.Size = UDim2.new(0, 0, 1, 0)

    local startTime = tick()

    local conn
    conn = RunService.Heartbeat:Connect(function()
        local elapsed = tick() - startTime
        local progress = math.clamp(elapsed / duration, 0, 1)
        loadingBar.Size = UDim2.new(progress, 0, 1, 0)

        loadingPercentLabel.Text = tostring(math.floor(progress * 100)) .. "%"

        -- Pulsing glow effect
        local pulse = 0.5 + 0.5 * math.sin(elapsed * 8)
        loadingBar.BackgroundColor3 = Color3.fromHSV(0.75, 1, pulse * 0.8 + 0.2)

        if progress >= 1 then
            conn:Disconnect()
            loadingBarContainer.Visible = false
            loadingPercentLabel.Visible = false
            callback()
        end
    end)
end

-- Show error message function (3 seconds visible)
local function showError(message)
    errorLabel.Text = message
    errorLabel.Visible = true
    task.delay(3, function()
        errorLabel.Visible = false
    end)
end

-- Show duping complete message function
local function showDupingComplete()
    dupingCompleteLabel.Visible = true
    dupingCompleteLabel.TextTransparency = 1

    TweenService:Create(dupingCompleteLabel, TweenInfo.new(0.3), {TextTransparency = 0}):Play()

    delay(3, function()
        TweenService:Create(dupingCompleteLabel, TweenInfo.new(0.5), {TextTransparency = 1}):Play()
        task.delay(0.5, function()
            dupingCompleteLabel.Visible = false
        end)
    end)
end
